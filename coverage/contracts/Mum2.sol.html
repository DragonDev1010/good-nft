	/**
	* @dev Holders of the Star NFTs can mint 1 free per Star they own.
	*/
	function holderMint(uint256[] memory _ids) public
	{
		Erequire(holderMintActive, "Holder mint is paused.");
		Erequire(_ids.length <= maxBatchAmount, "Can't mint that many.");
 
		for (uint256 i=0; i<_ids.length; i++)
		{
			Erequire(starContract.ownerOf(_ids[i]) == msg.sender, "You don't own this one.");
			Erequire(!usedStarIds[_ids[i]], "This one was already used for minting.");
		}
 
		_safeMint(msg.sender, _ids.length);
 
		for (uint256 i=0; i<_ids.length; i++)
		{
			usedStarIds[_ids[i]] = true;
			matchStarComp[_ids[i]] = holderMintedAmount + i;
		}
 
		holderMintedAmount += _ids.length;
	}
 
 
	/**
        * @dev Whitelisted wallets.
        */
	function whitelistMint(bytes32[] memory _proof, uint256 _num_tokens) public payable
	{
		Erequire(whitelistMintActive, "Whitelist mint is paused.");
		Erequire(msg.value == _num_tokens.mul(whitelistMintPrice), "Insufficient funds.");
		Erequire(mintedWhitelist[msg.sender] != true, "This wallet already minted");
 
		bytes32 leaf = keccak256(abi.encodePacked(msg.sender));
		Erequire(MerkleProof.verify(_proof, whitelistRoot, leaf), "Invalid proof.");
 
		_safeMint(msg.sender, _num_tokens);
 
		mintedWhitelist[msg.sender] = true;
		whitelistMintedAmount += _num_tokens;
	}
